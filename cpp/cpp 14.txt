#include <iostream>
using namespace std;
int main(){
int n;
cout<<"Enter a number to check it's divisibility by 5 and 3:";
cin>>n;
if(n%3==0 && n%5==0){
cout<<"divisible by both 3 and 5"<<endl;
}
else if(n%3==0){
cout<<"divisible by 3"<<endl;
}
else if(n%5==0){
cout<<"divisible  by 5"<<endl;
}
else{
cout<<"not divisible by both"<<endl;
}
return 0;
}
