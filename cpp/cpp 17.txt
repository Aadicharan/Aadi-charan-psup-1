#include <iostream>
using namespace std;

int main() {
    char ch;
    int a, b;

    cout << "Enter an operator (+, -, *, /): ";
    cin >> ch;

    cout << "Enter two numbers: ";
    cin >> a >> b;

    switch (ch) {
        case '+':
            cout << "Result: " << a + b << endl;
            break;
        case '-':
            cout << "Result: " << a - b << endl;
            break;
        case '*':
            cout << "Result: " << a * b << endl;
            break;
        case '/':
            if (b != 0)
                cout << "Result: " << a / b << endl;
            else
                cout << "Error: Division by zero!" << endl;
            break;
        default:
            cout << "Invalid operator!" << endl;
    }
